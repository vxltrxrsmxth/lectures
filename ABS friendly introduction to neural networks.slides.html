import math

def sigmoid(x):
  return round(1 / (1 + math.exp(-x)), 4)

def PerceptronANDInference(A, B):
    b = -50
    w1 = 30
    w2 = 30
    x1 = A
    x2 = B
    return sigmoid(w1*x1 + w2*x2 + b)
